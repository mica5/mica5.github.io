  • controllers and the "update" method
    • Elements that users can interact with that send messages that can be used to update the model
    • -- returns elements that users can interact with
-- that send messages that can be used to update
-- the model
buttons: List (Html Msg)
buttons =
  [
    button [ onClick Decrement ] [ text "-" ],
    button [ onClick Increment ] [ text "+" ]
  ]
    • Receive messages and use them to update the model
    • -- takes a message and a model and returns
-- a model that has the changes
update: Msg -> Model -> Model
update msg model =
  case msg of
    Increment -> model + 1
    Decrement -> model - 1

Elm Buttons Example in 5 minutes 
import Html exposing (div, button, text)
import Html.Events exposing (onClick)

main = Html.beginnerProgram {
  model=model,
  view=view,
  update=update
  }

type alias Model = {value: Int}
model = Model 0

type Msg =
    Noop
    | Increment
    | Decrement

view model =
    div [] [
        button [onClick Decrement] [text "-"],
        button [onClick Increment] [text "+"],
        text (toString model.value)
    ]

update msg model =
    case msg of
        Increment -> {model|value=model.value+1}
        Decrement -> {model|value=model.value-1}
        _ -> model
